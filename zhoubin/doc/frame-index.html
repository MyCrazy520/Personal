rabbitMQ安装

  1. 安装erlang 
    2. 					# yum -y install make gcc gcc-c++ kernel-devel m4 ncurses-devel openssl-devel unixODBC-devel
					# wget http://erlang.org/download/otp_src_20.3.tar.gz
					# tar -xvzf otp_src_19.0.tar.gz
					# ./configure --prefix=/usr/local/erlang --with-ssl -enable-threads -enable-smmp-support -enable-kernel-poll --enable-hipe --without-javac
					# make
					# make install

					# ERLANG_HOME=/usr/local/erlang # 配置profile
					# PATH=$PATH:$JAVA_HOME/bin:$ERLANG_HOME/bin  # 配置profile
					# source /etc/profile   # 使其生效

					# erl 检验erl
					# > Eshell V9.3  (abort with ^G)
  2. 安装 rabbitmq 
    3. 					# wget https://github.com/rabbitmq/rabbitmq-server/releases/download/v3.7.10/rabbitmq-server-generic-unix-3.7.10.tar.xz
					# xz -d rabbitmq-server-generic-unix-3.7.10.tar.xz
					# tar -xvf rabbitmq-server-generic-unix-3.7.10.tar
  3. 启动 rabbitmq 
    4. 					# cd rabbitmq_server-3.7.10/sbin
					# ./rabbitmq-plugins enable rabbitmq_management  #启用web管理界面
					# ./rabbitmq-server -detached  #启动
					# ./rabbitmqctl add_user zhoubin Zhoubin77! #添加用户
					# ./rabbitmqctl set_user_tags zhoubin administrator #设置权限
					# 浏览器访问:ip:15672
  4. 管理rabbitmq
    5. 					# ./rabbitmq-server –detached  #启动
					# ./rabbitmqctl stop #关闭

					#插件管理
					# ./rabbitmq-plugins enable xxx #开启某个插件
					# ./rabbitmq-plugins disable xxx #关闭某个插件

					# virtual_host  管理
					# ./rabbitmqctl add_vhost xxx  #新建 virtual_host
					# ./rabbitmqctl delete_vhost xxx  #撤销 virtual_host

					# 用户管理
					# ./rabbitmqctl add_user xxx pwd #新建用户
					# ./rabbitmqctl delete_user xxx #删除用户
					# ./rabbimqctl change_password username newpassword #改密码
					# ./rabbitmqctl set_user_tags username tag #设置用户角色;Tag可以为 administrator,monitoring, management

					#权限管理
					# ./rabbitmqctl set_permissions [-p vhostpath] {user} {conf} {write}{read}
					# vhostpath Vhost路径
					# user用户名
					# conf一个正则表达式match哪些配置资源能够被该用户访问。
					# write 一个正则表达式match哪些配置资源能够被该用户读。
					# read 一个正则表达式match哪些配置资源能够被该用户访问
					#eg: ./rabbitmqctl  set_permissions -p "/" admin ".*" ".*" ".*"

					# 常用命令
					# ./rabbitmqctl status #查看状态
					# ./rabbitmqctl list_queues #查看所有队列信息
					# ./rabbitmqctl stop_app #关闭应用
					# ./rabbitmqctl start_app #启动应用
					# ./rabbitmqctl reset #清除所有队列

redis安装

  1. 下载redis安装包
    2. 					# wget http://download.redis.io/releases/redis-4.0.9.tar.gz
  2. 解压安装包
    3. 					# tar -zxvf redis-4.0.9.tar.gz -C /usr/local/
  3. 安装gcc依赖:
    4. 					# 先通过gcc -v是否有安装gcc，如果没有安装，执行命令sudo yum install -y gcc
  4. cd到redis的解压目录下，并执行:
    5. 					# cd /usr/local/redis-4.0.9/
  5. 编译安装:
    6. 					 make MALLOC=libc
					 cd src && make install
  6. 测试是否安装成功
    7. 					 cd /usr/local/redis-4.0.9/src/
					 ./redis-server
  7. 配置redis
    8. 					# 修改 /usr/local/redis-4.0.9/redis.conf：    daemonize no   将值改为yes 保存退出
					# 修改 /usr/local/redis-4.0.9/redis.conf：    logfile "/usr/local/redis/logs/log"
					# 修改 /usr/local/redis-4.0.9/redis.conf：    dir /usr/local/redis/data

					 ./redis-server /usr/local/redis-4.0.9/redis.conf #指定redis.conf文件启动
  8. 设置redis远程连接
    9. 					# 因为redis默认设置允许本地连接，将redis.conf中将bind 127.0.0.1 改为bind 0.0.0.0或者注释该行或者加入服务器IP
					# 另外，阿里云ECS有一个安全组，找到并添加规则允许6379端口访问
  9. 设置redis连接密码
    10. 					# requirepass yourpassword  #在redis.conf中搜索requirepass这一行，然后在合适的位置添加配置
					# 设置完成后执行/usr/local/bin/redis-server /usr/local/redis-4.0.6/redis.conf 更新配置
  10. 设置开机自启动
    11. 					#加入密码
					vim /usr/local/redis-4.0.9/redis.conf
					$CLIEXEC -a "zhoubin" -p $REDISPORT shutdown
				
    					 mkdir /etc/redis
					 cp /usr/local/redis-4.0.9/redis.conf /etc/redis/6379.conf
					 cp /usr/local/redis-4.0.9/utils/redis_init_script /etc/init.d/redisd
					# 切换到/etc/init.d目录下
					# vim编辑redisd文件，在第一行加入如下两行注释，保存退出
					# chkconfig:   2345 90 10
					# description:  Redis is a persistent key-value database

					 chkconfig redisd on #执行自启命令

					#以后可以直接已服务的形式启动和关闭redis了
					 service redisd start
					 service redisd stop
  11. service redis stop报错解决方法
    12. 					./src/redis-cli -p 6379 -a zhoubin shutdown
				
    					#修改redis服务脚本，加入如下所示的红色授权信息即可：
					vi /etc/init.d/redisd
					$CLIEXEC -a "zhoubin" -p $REDISPORT shutdown

					makdir /usr/local/nginx
					./configure --prefix=/usr/local/nginx
					make

  2. nginx启动
    3. 					cd /usr/local/nginx
					./sbin/nginx #配置启动
					nginx -s reload|reopen|stop|quit  #重新加载配置|重启|停止|退出 nginx

  3. nginx配置
